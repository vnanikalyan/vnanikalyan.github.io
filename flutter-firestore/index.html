Future<Map> getData(<CollectionName>, <DocumentName>) async {
    var x = await Firestore.instance.collection(<CollectionName>)
                  .document(<DocumentName>).get()
                  .then((document){
                                    //return document.data;
                                    return document;
                                  });

    print('Data : ${x.data}');
    return x.data;
}

When you want to retrieve data based on CollectionName and FieldName

Future<List> getEvents(context, CollectionName, FieldName) async {
    var x = await Firestore.instance.collection(CollectionName).where(FieldName, isEqualTo: state.churchId).getDocuments();
    int length = x.documents.length;
    List list = new List(length);
    
    for(int i=0; i<length; i++){
        List[i] = x.documents[i].data
    }
    return list;
}
